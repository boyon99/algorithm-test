# 좌표 정렬하기

### 문제

https://www.acmicpc.net/problem/11650

<hr/>

### 정답

```js
let [N, ...XY] = require("fs")
  .readFileSync(process.platform === "linux" ? "/dev/stdin" : "./input.txt")
  .toString()
  .trim()
  .split("\n");

let Arr = new Array();

// 이차원 배열로 저장
for (let i = 0; i < N; i++) {
  const [x, y] = XY[i].split(" ").map(Number);
  Arr.push([x, y]);
}

// 배열 정렬하기
let SortArr = Arr.sort((a, b) => {
  if (a[0] === b[0]) {
    return a[1] - b[1];
  } else {
    return a[0] - b[0];
  }
});

console.log(SortArr.join("\n").replace(/,/g, " "));
```
